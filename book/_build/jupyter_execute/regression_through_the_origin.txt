import pandas as pd
import numpy as np
import matplotlib.pyplot as plt


dat = pd.read_csv("https://raw.githubusercontent.com/bcaffo/ds4bme_intro/master/data/oasis.csv")
dat.head()

x = dat.T2
y = dat.PD
plt.plot(x, y, 'o')

x = x - np.mean(x)
y = y - np.mean(y)
plt.plot(x, y, 'o')

b = sum(y * x) / sum(x ** 2 )
b

plt.plot(x, y, 'o')
t = np.array([-1.5, 2.5])
plt.plot(t, t * b)
